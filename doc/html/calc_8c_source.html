34  if(!bounds_relative)

41  if(boundary_size)
42  TYM_POS_REF(res, RATIO, axis) = (double)TYM_POS_REF(res, CHARFIELD, axis) / boundary_size;
43  }

48  struct tym_absolute_position_rectangle res;

50  tym_i_calc_absolut_position(&res.edge[edge], &tym_i_bounds, &super_position->edge[edge], true);
51  if( TYM_RECT_POS_REF(res, CHARFIELD, TYM_LEFT) >= TYM_RECT_POS_REF(res, CHARFIELD, TYM_RIGHT )

53  ){
54  memset(&res, 0, sizeof(res));
