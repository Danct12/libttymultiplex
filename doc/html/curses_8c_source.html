172 static int pane_scroll_region(struct tym_i_pane_internal* pane, int n, unsigned top, unsigned bottom){
173  if(n == 0)
174  return 0;
175  if(top >= bottom)
176  return -1;
177  struct curses_backend_pane* cbp = pane->backend;
178  struct curses_screen_state* cscreen = &cbp->screen[pane->current_screen];
179  long w = TYM_RECT_SIZE(pane->absolute_position, CHARFIELD, TYM_AXIS_HORIZONTAL);
180  long h = TYM_RECT_SIZE(pane->absolute_position, CHARFIELD, TYM_AXIS_VERTICAL);
181  if(top >= h)
182  return -1;
183  if(bottom > h)
184  bottom = h;
185  unsigned long m = bottom - top;
186  tym_i_debug("moving scrolling region: %u %u %lu %d\n",top,bottom,m,n);
187  if(m <= (unsigned)abs(n)){
188  return tym_i_backend->pane_erase_area(
189  pane,
190  (struct tym_i_cell_position){ .x=0, .y=top },
191  (struct tym_i_cell_position){ .x=w, .y=bottom },
192  true,
193  (struct tym_i_character_format){0}
194  );
195  }else{
196  WINDOW* region = subpad(cscreen->window, m, w, top, 0);
197  if(!region)
198  return -1;
199  scrollok(region, true);
200  int res = wscrl(region, n) == OK ? 0 : -1;
201  scrollok(region, false);
