26 bool tym_i_character_is_utf8(struct tym_i_character character){
27  uint8_t gl = character.charset_selection;
28  uint8_t gr = character.charset_selection >> 8;
29  if(character.not_utf8)
30  return false;
31  if(gl != 0 || gr != 0)
32  return false;
33  if(character.charset_g[gl] || character.charset_g[gr])
34  return false;
35  return true;

270 int tym_i_pane_set_cursor_position(
271  struct tym_i_pane_internal* pane,
272  enum tym_i_scp_position_mode pm_x, long long x,
273  enum tym_i_scp_scrolling_mode smm_y, enum tym_i_scp_position_mode pm_y, long long y,
274  enum tym_i_scp_scroll_region_behaviour srb, bool allow_cursor_on_right_edge
275 ){
276  struct tym_i_pane_screen_state* screen = &pane->screen[pane->current_screen];
277  unsigned w = TYM_RECT_SIZE(pane->absolute_position, CHARFIELD, TYM_AXIS_HORIZONTAL);
278  unsigned h = TYM_RECT_SIZE(pane->absolute_position, CHARFIELD, TYM_AXIS_VERTICAL);
279  bool scroll_region_valid = screen->scroll_region_top < screen->scroll_region_bottom && screen->scroll_region_top < h;
280  bool origin_mode = screen->origin_mode && scroll_region_valid;
281  unsigned sr_top = scroll_region_valid ? screen->scroll_region_top : 0;
282  unsigned sr_bottom = scroll_region_valid ? screen->scroll_region_bottom : h;
283 
285 
286  switch(pm_x){
287  case TYM_I_SCP_PM_ABSOLUTE: break;
288  case TYM_I_SCP_PM_RELATIVE: x += screen->cursor.x; break;
289  case TYM_I_SCP_PM_ORIGIN_RELATIVE: break;
290  }
291 
292  switch(pm_y){
293  case TYM_I_SCP_PM_ABSOLUTE: break;
294  case TYM_I_SCP_PM_RELATIVE: y += screen->cursor.y; break;
295  case TYM_I_SCP_PM_ORIGIN_RELATIVE: {
296  if(origin_mode)
297  y += screen->scroll_region_top;
298  } break;
299  }
300 
302 
303  long long top = 0;
304  long long bottom = h;
305  long long left = 0;
306  long long right = w;
307 
308  switch(srb){
309  case TYM_I_SCP_SCROLLING_REGION_IRRELEVANT: break;
310  case TYM_I_SCP_SCROLLING_REGION_UNCROSSABLE: {
311  if( screen->cursor.y >= sr_top && y < sr_top )
312  goto set_scrolling_region_boundary;
313  if( screen->cursor.y < (long long)sr_bottom && y >= (long long)sr_bottom )
314  goto set_scrolling_region_boundary;
315  if( screen->cursor.y < (long long)sr_bottom && screen->cursor.y >= (long long)sr_top )
316  goto set_scrolling_region_boundary;
317  } break;
318  case TYM_I_SCP_SCROLLING_REGION_LOCKIN_IN_ORIGIN_MODE: {
319  if(origin_mode)
320  goto set_scrolling_region_boundary;
321  } break;
322  set_scrolling_region_boundary: {
323  if(!scroll_region_valid)
324  break;
325  top = sr_top;
326  bottom = sr_bottom;
327  } break;
328  }
329 
331 
332  switch(smm_y){
333  case TYM_I_SCP_SMM_NO_SCROLLING: break;
334  case TYM_I_SCP_SMM_SCROLL_FORWARD_ONLY: {
335  if(y >= bottom)
336  tym_i_scroll_def_scrolling_region(pane, top, bottom, y - bottom + 1);
337  } break;
338  case TYM_I_SCP_SMM_SCROLL_BACKWARD_ONLY: {
339  if(y < top)
340  tym_i_scroll_def_scrolling_region(pane, top, bottom, y - top);
341  } break;
342  case TYM_I_SCP_SMM_UNRESTRICTED_SCROLLING: {
343  if(y < top)
344  tym_i_scroll_def_scrolling_region(pane, top, bottom, y - top);
345  if(y >= bottom)
346  tym_i_scroll_def_scrolling_region(pane, top, bottom, y - bottom + 1);
347  }; break;
348  }
349 
351 
352  if(y >= bottom)
353  y = bottom - 1;
354  if(y < top)
355  y = top;
356  if(x >= right)
357  x = right - !allow_cursor_on_right_edge;
358  if(x < left)
359  x = left;
360 
361  screen->cursor.y = y;
362  screen->cursor.x = x;
363 
364  tym_i_pane_update_cursor(pane);
365 
366  return 0;
367 }
