105 void* tym_i_main(void* ptr){
106  (void)ptr;
107  while(true){
108  int ret = poll(tym_i_poll_fds, tym_i_poll_fdn, -1);
109  pthread_mutex_lock(&tym_i_lock);
110  if(ret == -1){
111  if(errno == EINTR)
112  goto cont;
113  perror("poll failed");
114  goto error;
115  }
116  if(tym_i_poll_fds[SPF_POLLCTLFD].revents & (POLLHUP|POLLERR|POLLNVAL)){
117  pthread_mutex_unlock(&tym_i_lock);
118  break;
119  }
120  if(tym_i_poll_fds[SPF_POLLCTLFD].revents & POLLIN){
121  struct tym_i_poll_ctl ctl = {0};
122  ssize_t s;
123  while((s=read(tym_i_pollctl[0], &ctl, sizeof(ctl))) == -1 && errno == EINTR);
124  if(s == -1){
125  perror("read failed");
126  goto error;
127  }
128  switch(ctl.action){
129  case TYM_PC_ADD: {
130  tym_i_pollfd_add_sub((struct pollfd){
131  .fd = ctl.data.add.fd,
132  .events = POLLIN | POLLHUP
133  });
134  } break;
135  case TYM_PC_REMOVE: {
136  for(size_t i=SPF_COUNT; i<tym_i_poll_fdn; i++)
137  if(tym_i_poll_fds[i].fd == ctl.data.remove.fd){
138  tym_i_pollfd_remove_sub(i);
139  break;
140  }
141  } break;
142  }
143  }else if(tym_i_poll_fds[SPF_POLLCTLFD].revents){
144  pthread_mutex_unlock(&tym_i_lock);
145  break;
146  }
147  if(tym_i_poll_fds[SPF_SIGNALFD].revents & POLLIN){
148  struct signalfd_siginfo info;
149  ssize_t s = read(tym_i_sfd, &info, sizeof(info));
150  if(s != sizeof(info)){
151  perror("read failed");
152  goto error;
153  }
154  switch(info.ssi_signo){
155  case SIGWINCH: tym_i_pane_update_size_all(); break;
156  case SIGSTOP:
157  case SIGTSTP:
158  case SIGINT: {
159  if(!tym_i_focus_pane)
160  break;
161 #if defined(TIOCSIG)
162  ioctl(tym_i_focus_pane->master, TIOCSIG, info.ssi_signo);
163 #elif defined(TIOCSIGNAL)
164  if(tym_i_focus_pane)
165  ioctl(tym_i_focus_pane->master, TIOCSIGNAL, info.ssi_signo);
166 #endif
167  } break;
168  }
169  }else if(tym_i_poll_fds[SPF_SIGNALFD].revents){
170  pthread_mutex_unlock(&tym_i_lock);
171  break;
172  }
173  if(tym_i_poll_fds[SPF_TERMINPUT].revents & (POLLHUP|POLLERR|POLLNVAL)){
174  pthread_mutex_unlock(&tym_i_lock);
175  break;
176  }else if(tym_i_poll_fds[SPF_TERMINPUT].revents & POLLIN){
177  // TODO: abstract this away into the backend too
178  int c = getch();
179  if(c != ERR)
180  switch(c){
181  case KEY_MOUSE: {
182  MEVENT event;
183  if(getmouse(&event) != OK)
184  break;
185  for(struct tym_i_pane_internal* it=tym_i_pane_list_start; it; it=it->next){
186  int left = TYM_RECT_POS_REF(it->absolute_position, CHARFIELD, TYM_LEFT );
187  int right = TYM_RECT_POS_REF(it->absolute_position, CHARFIELD, TYM_RIGHT );
188  int top = TYM_RECT_POS_REF(it->absolute_position, CHARFIELD, TYM_TOP );
189  int bottom = TYM_RECT_POS_REF(it->absolute_position, CHARFIELD, TYM_BOTTOM);
190  if( left > (int)event.x || right <= (int)event.x || top > (int)event.y || bottom <= (int)event.y )
191  continue;
192  unsigned x = event.x - left;
193  unsigned y = event.y - top;
194  tym_i_pane_focus(it);
195  if(event.bstate & (BUTTON1_RELEASED | BUTTON2_RELEASED | BUTTON3_RELEASED)){
196  tym_i_pts_send_mouse_event(it, TYM_BUTTON_RELEASED, (struct tym_i_cell_position){.x=x, .y=y});
197  }
198  if(event.bstate & BUTTON1_PRESSED){
199  tym_i_pts_send_mouse_event(it, TYM_BUTTON_LEFT_PRESSED, (struct tym_i_cell_position){.x=x, .y=y});
200  }
201  if(event.bstate & BUTTON2_PRESSED){
202  tym_i_pts_send_mouse_event(it, TYM_BUTTON_MIDDLE_PRESSED, (struct tym_i_cell_position){.x=x, .y=y});
203  }
204  if(event.bstate & BUTTON3_PRESSED){
205  tym_i_pts_send_mouse_event(it, TYM_BUTTON_RIGHT_PRESSED, (struct tym_i_cell_position){.x=x, .y=y});
206  }
207  break;
208  }
209  } break;
210  case KEY_ENTER: tym_i_pts_send_key(tym_i_focus_pane, TYM_KEY_ENTER); break;
211  case KEY_UP : tym_i_pts_send_key(tym_i_focus_pane, TYM_KEY_UP); break;
212  case KEY_DOWN : tym_i_pts_send_key(tym_i_focus_pane, TYM_KEY_DOWN); break;
213  case KEY_RIGHT: tym_i_pts_send_key(tym_i_focus_pane, TYM_KEY_RIGHT); break;
214  case KEY_LEFT : tym_i_pts_send_key(tym_i_focus_pane, TYM_KEY_LEFT); break;
215  case KEY_BACKSPACE: tym_i_pts_send_key(tym_i_focus_pane, TYM_KEY_BACKSPACE); break;
216  case KEY_HOME: tym_i_pts_send_key(tym_i_focus_pane, TYM_KEY_HOME); break;
217  case KEY_END: tym_i_pts_send_key(tym_i_focus_pane, TYM_KEY_END); break;
218  case KEY_DC: tym_i_pts_send_key(tym_i_focus_pane, TYM_KEY_DELETE); break;
219  default: tym_i_pts_send_key(tym_i_focus_pane, c); break;
220  }
221  }else if(tym_i_poll_fds[SPF_TERMINPUT].revents){
222  pthread_mutex_unlock(&tym_i_lock);
223  break;
224  }
225  for(size_t i=SPF_COUNT; i<tym_i_poll_fdn; i++){
226  if(!tym_i_poll_fds[i].revents)
227  continue;
228  if(tym_i_poll_fds[i].revents & (POLLERR|POLLNVAL)){
229  tym_i_pollfd_remove_sub(i);
230  }else if(tym_i_poll_fds[i].revents & POLLIN){
231  static char buf[256];
232  ssize_t ret;
233  do {
234  ret = read(tym_i_poll_fds[i].fd, buf, sizeof(buf));
235  } while(ret == -1 && errno == EINTR);
236  if(ret == -1 && errno == EBADF)
237  tym_i_pollfd_remove_sub(i);
238  if(ret > 0)
239  for(struct tym_i_pane_internal* it=tym_i_pane_list_start; it; it=it->next){
240  if(it->master != tym_i_poll_fds[i].fd)
241  continue;
242  for(size_t i=0; i<(size_t)ret; i++)
243  tym_i_pane_parse(it, buf[i]);
244  tym_i_backend->pane_refresh(it);
245  break;
246  }
247  }else{
248  tym_i_pollfd_remove_sub(i);
249  }
250  }
251  cont:
252  pthread_mutex_unlock(&tym_i_lock);
253  }
254  pthread_mutex_lock(&tym_i_lock);
255  while(tym_i_poll_fdn)
256  tym_i_pollfd_remove_sub(0);
257  tym_i_backend->cleanup();
258  tym_i_binit = INIT_STATE_SHUTDOWN;
259  pthread_mutex_unlock(&tym_i_lock);
260  return 0;
261 error:
262  pthread_mutex_unlock(&tym_i_lock);
263  abort();
264 }
