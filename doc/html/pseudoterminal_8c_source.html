24 int tym_i_pts_send_key(struct tym_i_pane_internal* pane, int_least16_t key){
25  if(!pane){
26  errno = EINVAL;
27  return -1;
28  }
29  struct tym_i_pane_screen_state* screen = &pane->screen[pane->current_screen];
30  #define CS(X) case TYM_KEY_ ## X:
31  switch((enum tym_special_key)key){
32  CS(UP) CS(DOWN) CS(RIGHT) CS(LEFT) CS(HOME) CS(END) {
33  switch(screen->cursor_key_mode){
34  case TYM_I_KEYPAD_MODE_NORMAL: switch(key){
35  case TYM_KEY_UP : return tym_i_pts_send(pane, S(CSI "A"));
36  case TYM_KEY_DOWN : return tym_i_pts_send(pane, S(CSI "B"));
37  case TYM_KEY_RIGHT: return tym_i_pts_send(pane, S(CSI "C"));
38  case TYM_KEY_LEFT : return tym_i_pts_send(pane, S(CSI "D"));
39  case TYM_KEY_HOME : return tym_i_pts_send(pane, S(CSI "H"));
40  case TYM_KEY_END : return tym_i_pts_send(pane, S(CSI "F"));
41  default: return -1;
42  } break;
43  case TYM_I_KEYPAD_MODE_APPLICATION: switch(key){
44  case TYM_KEY_UP : return tym_i_pts_send(pane, S(SS3 "A"));
45  case TYM_KEY_DOWN : return tym_i_pts_send(pane, S(SS3 "B"));
46  case TYM_KEY_RIGHT: return tym_i_pts_send(pane, S(SS3 "C"));
47  case TYM_KEY_LEFT : return tym_i_pts_send(pane, S(SS3 "D"));
48  case TYM_KEY_HOME : return tym_i_pts_send(pane, S(SS3 "H"));
49  case TYM_KEY_END : return tym_i_pts_send(pane, S(SS3 "F"));
50  default: return -1;
51  } break;
52  }
53  } break;
54  case TYM_KEY_PAGE_UP: return tym_i_pts_send(pane, S(CSI "5~"));
55  case TYM_KEY_PAGE_DOWN: return tym_i_pts_send(pane, S(CSI "6~"));
56  case TYM_KEY_ENTER: return tym_i_pts_send(pane, S("\r"));
57  case TYM_KEY_TAB: break;
58  case TYM_KEY_BACKSPACE: break;
59  case TYM_KEY_ESCAPE: break;
60  case TYM_KEY_DELETE: return tym_i_pts_send(pane, S(CSI "3~"));
61  }
62  if(key >= 0 && key < 0x100)
63  return tym_i_pts_send(pane, 1, (char[]){key});
64 #undef CS
65  return -1;
66 }

80 int tym_i_pts_send_mouse_event(struct tym_i_pane_internal* pane, enum tym_button button, struct tym_i_cell_position pos){
81  char buf[64] = {0};
82  int len = 0;
83  bool motion = pane->last_mouse_event_pos.x != pos.x || pane->last_mouse_event_pos.y != pos.y;
84  if(pane->last_button != button)
85  motion = false;
86  pane->last_mouse_event_pos = pos;
87  switch(pane->mouse_mode){
88  case MOUSE_MODE_X10:
89  if(button == TYM_BUTTON_RELEASED)
90  break;
91  case MOUSE_MODE_NORMAL:
92  if(pane->last_button == button)
93  break;
94  case MOUSE_MODE_BUTTON:
95  if(pane->last_button == button && button == TYM_BUTTON_RELEASED)
96  break;
97  case MOUSE_MODE_ANY: {
98  if(pos.x>254 || pos.y>254)
99  break; // Can't encode coordinate in a byte each, overflow!!! (TODO: implement stuff like utf8 encoded positions)
100  unsigned char cb = 32;
101  if(motion)
102  cb += 32; // Motion indicator
103  len = snprintf(buf, sizeof(buf), CSI "M%c%c%c", cb+button, (unsigned char)(32+pos.x+1), (unsigned char)(32+pos.y+1));
104  } break;
105  case MOUSE_MODE_OFF: break;
106  }
107  pane->last_button = button;
108  if(len)
109  return tym_i_pts_send(pane, len, buf);
110  return 0;
111 }
