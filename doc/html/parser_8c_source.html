152 int command_sequence_comparator_ch(char ca, char cb){
153  if(ca == cb)
154  return 0;
155  if((ca < ' ') == (cb < ' '))
156  return (ca < cb) ? -1 : 1;
157  return ca < ' ' ? 1 : -1;
158 }

160 int command_sequence_comparator_str(const char*restrict ia, const char*restrict ib){
161  for( ; *ia && *ib; ia++, ib++ ){
162  int ret = command_sequence_comparator_ch(*ia, *ib);
163  if(ret)
164  return ret;

166  if(*ia) return 1;

198 static enum specialmatch_result specialmatch(unsigned char a, unsigned char b){
199  switch(a){
200  case '\1': return SM_MATCH_SINGLE;
201  case '\2': return (b >= '0' && b <= '9') || b == ';' ? SM_MATCH_CONTINUE : SM_NO_MATCH;
202  case '\3': return (b >= '0' && b <= '9') ? SM_MATCH_CONTINUE : SM_NO_MATCH;
203  case '\4': return b >= ' ' ? SM_MATCH_CONTINUE : SM_NO_MATCH;
204  default: return SM_NO_MATCH;
205  }
206 }

218 bool control_character(struct tym_i_pane_internal* pane, unsigned char c){
219  struct tym_i_pane_screen_state* screen = &pane->screen[pane->current_screen];
220  unsigned y = screen->cursor.y;
221  unsigned x = screen->cursor.x;
222  unsigned w = TYM_RECT_SIZE(pane->absolute_position, CHARFIELD, TYM_AXIS_HORIZONTAL);
223  unsigned h = TYM_RECT_SIZE(pane->absolute_position, CHARFIELD, TYM_AXIS_VERTICAL);
224  if(x >= w)
225  x = w;
226  if(y >= h)
227  y = h;
228  switch(c){
229  case '\b': if(x) x -= 1; break;
230  case '\r': x = 0; break;
231  case '\t': x = x / 8 * 8 + 8; break;
232  case '\v': y += 1; break;
233  case '\n': y += 1; break; // TODO: implement automatic new line
234  case 0x0E /*SO*/: tym_i_invoke_charset(pane, TYM_I_CHARSET_SELECTION_GL_G1 | TYM_I_CHARSET_SELECTION_GR_G1); break;
235  case 0x0F /*SI*/: tym_i_invoke_charset(pane, TYM_I_CHARSET_SELECTION_GL_G0 | TYM_I_CHARSET_SELECTION_GR_G0); break;
236  }
237  if(c >= ' ')
238  return false;
239  tym_i_debug("Control character: x%02X\n",(int)c);
240  tym_i_pane_set_cursor_position( pane,
241  TYM_I_SCP_PM_ABSOLUTE, x,
242  TYM_I_SCP_SMM_SCROLL_FORWARD_ONLY, TYM_I_SCP_PM_ABSOLUTE, y,
243  TYM_I_SCP_SCROLLING_REGION_UNCROSSABLE, false
244  );
245  return true;
246 }

257 void print_character(struct tym_i_pane_internal* pane, const struct tym_i_character character){
258  if(tym_i_character_is_utf8(character) && !character.data.utf8.count)
259  return;
260  struct tym_i_pane_screen_state* screen = &pane->screen[pane->current_screen];
261  unsigned y = screen->cursor.y;
262  unsigned x = screen->cursor.x;
263  unsigned w = TYM_RECT_SIZE(pane->absolute_position, CHARFIELD, TYM_AXIS_HORIZONTAL);
264  unsigned h = TYM_RECT_SIZE(pane->absolute_position, CHARFIELD, TYM_AXIS_VERTICAL);
265  if(x >= w)
266  x = w;
267  if(y >= h)
268  y = h;
269  const char* sequence = 0;
270  if(tym_i_character_is_utf8(character)){
271  sequence = (char*)character.data.utf8.data;
272  }else{
273  bool codepage = !!(character.data.byte & 0x80);
274  uint8_t index = character.data.byte & 0x7F;
275  uint8_t gl = character.charset_selection;
276  uint8_t gr = character.charset_selection >> 8;
277  uint8_t g = codepage ? gr : gl;
278  if(g >= TYM_I_G_CHARSET_COUNT || index < ' '){
279  sequence = (char*)UTF8_INVALID_SYMBOL.data.utf8.data;
280  }else{
281  uint8_t charset = character.charset_g[g];
282  if(charset >= TYM_I_CHARSET_GENERIC_COUNT){
283  sequence = (char*)UTF8_INVALID_SYMBOL.data.utf8.data;
284  }else{
285  sequence = (char*)tym_i_translation_table[charset].table[index-' '];
286  }
287  }
288  }
289  if(x >= w){
290  x = 0;
291  if(!screen->wraparound_mode_off)
292  y += 1;
293  }
294  if(sequence)
295  tym_i_backend->pane_set_character(pane, (struct tym_i_cell_position){.x=x,.y=y}, screen->character_format, strlen(sequence), sequence, screen->insert_mode);
296  x += 1;
297  tym_i_pane_set_cursor_position( pane,
298  TYM_I_SCP_PM_ABSOLUTE, x,
299  TYM_I_SCP_SMM_SCROLL_FORWARD_ONLY, TYM_I_SCP_PM_ABSOLUTE, y,
300  TYM_I_SCP_SCROLLING_REGION_UNCROSSABLE, true
301  );
302 }

334 void tym_i_pane_parse(struct tym_i_pane_internal* pane, unsigned char c){
335 // tym_i_debug("tym_i_pane_parse %.2X\n", c);
336  if(tym_i_character_is_utf8(pane->character))
337  if( pane->character.data.utf8.count )
338  if(print_character_update(pane, c))
339  return;
340  unsigned short n = pane->sequence.length;
341  unsigned short index = pane->sequence.index;
342  if(n >= TYM_I_MAX_SEQ_LEN)
343  goto escape_abort;
344  if(!pane->sequence.length){
345  pane->sequence.seq_opt_min = 0;
346  pane->sequence.seq_opt_max = tym_i_command_sequence_map_count-1;
347  }
348  ssize_t min = pane->sequence.seq_opt_min;
349  ssize_t max = pane->sequence.seq_opt_max;
350  if( pane->sequence.last_special_match_continue && !specialmatch(tym_i_command_sequence_map[min].sequence[index], c) )
351  index++;
352  while( max >= min && ( tym_i_command_sequence_map[min].length <= index || (
353  tym_i_command_sequence_map[min].length > index
354  && command_sequence_comparator_ch(tym_i_command_sequence_map[min].sequence[index], c) < 0
355  && !specialmatch(tym_i_command_sequence_map[min].sequence[index], c)
356  ))) min++;
357  if( max >=min && tym_i_command_sequence_map[min].sequence[index] >= ' ' && command_sequence_comparator_ch(tym_i_command_sequence_map[min].sequence[index], c) > 0)
358  while( max >= min && (
359  tym_i_command_sequence_map[min].length <= index // Order sensitive!!
360  || tym_i_command_sequence_map[min].sequence[index] >= ' '
361  || !specialmatch(tym_i_command_sequence_map[min].sequence[index], c)
362  )) min++;
363  bool esm = false;
364  if(max >= min && tym_i_command_sequence_map[min].length > index)
365  if(specialmatch(tym_i_command_sequence_map[min].sequence[index], c))
366  esm = true;
367  while( max >= min && tym_i_command_sequence_map[max].length > index
368  && command_sequence_comparator_ch(tym_i_command_sequence_map[max].sequence[index], c) > 0
369  && (!esm||!specialmatch(tym_i_command_sequence_map[min].sequence[index], c))
370  ) max--;
371 /* if(max < min)
372  fprintf(stderr,"%*s %c\n",(int)pane->sequence.length,pane->sequence.buffer, c);*/
373  if(max < min || index >= tym_i_command_sequence_map[min].length || index >= tym_i_command_sequence_map[max].length){
374  if(control_character(pane, c)){
375  return;
376  }else{
377  goto escape_abort;
378  }
379  }
380  pane->sequence.seq_opt_min = min;
381  pane->sequence.seq_opt_max = max;
382  if(max >= min && tym_i_command_sequence_map[min].sequence[index] < ' '){
383  switch(tym_i_command_sequence_map[min].sequence[index]){
384  case '\1': {
385  if(++pane->sequence.integer_count >= TYM_I_MAX_INT_COUNT)
386  goto escape_abort;
387  pane->sequence.integer[pane->sequence.integer_count-1] = c;
388  } break;
389  case '\2': {
390  if(!pane->sequence.integer_count)
391  pane->sequence.integer_count = 1;
392  if(c == ';'){
393  if(++pane->sequence.integer_count >= TYM_I_MAX_INT_COUNT)
394  goto escape_abort;
395  }else if(c >= '0' && c <= '9'){
396  pane->sequence.integer[pane->sequence.integer_count-1] *= 10;
397  pane->sequence.integer[pane->sequence.integer_count-1] += c - '0';
398  }else goto escape_abort;
399  } break;
400  }
401  }
402  if(min == max && index+1 == tym_i_command_sequence_map[min].length){
403  const struct tym_i_command_sequence* command = tym_i_command_sequence_map + min;
404  if(command->callback){
405  if((*command->callback)(pane) == -1){
406  int err = errno;
407  tym_i_debug("- ");
408  tym_i_debug_sequence_params(command, &pane->sequence);
409  tym_i_debug(": %d %s\n", err, strerror(err));
410  if(err == ENOENT)
411  goto escape_abort;
412  }else{
413  tym_i_pane_update_cursor(pane);
414  tym_i_debug("+ ");
415  tym_i_debug_sequence_params(command, &pane->sequence);
416  tym_i_debug("\n");
417  }
418  }else{
419  tym_i_debug("? ");
420  tym_i_debug_sequence_params(command, &pane->sequence);
421  tym_i_debug("\n");
422  }
423  reset_sequence(&pane->sequence);
424  return;
425  }
426  pane->sequence.buffer[n] = c;
427  pane->sequence.length = n + 1;
428  if( specialmatch(tym_i_command_sequence_map[min].sequence[index], c) == SM_MATCH_CONTINUE ){
429  pane->sequence.last_special_match_continue = true;
430  }else{
431  pane->sequence.index = index + 1;
432  }
433  return;
434 escape_abort:;
435  char fc = pane->sequence.length ? pane->sequence.buffer[0] : c;
436  switch(fc){
437  case '\x1B': print_character_update(pane, '^'); break;
438  default: print_character_update(pane, fc); break;
439  }
440  if(pane->sequence.length){
441  tym_i_debug("%.*s%c %zd %zd\n",(int)pane->sequence.length,pane->sequence.buffer,c, pane->sequence.seq_opt_min, pane->sequence.seq_opt_max);
442  reset_sequence(&pane->sequence);
443  for(unsigned short i=1; i<n; i++)
444  tym_i_pane_parse(pane, pane->sequence.buffer[i]);
445  tym_i_pane_parse(pane, c);
446  }
447  tym_i_pane_update_cursor(pane);
448  return;
449 }
